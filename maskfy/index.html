Exemple

Setup

HTML

<input class="page--input mask-by-class" type="text" value="">
or
<input class="page--input" data-mask="999.999.999-99" type="text" placeholder="CPF (Ex.: 999.999.999-99)">

JS Library

<script src="path/to/ibaro.maskfy.min.js"></script>

JS init by Attribute HTML

<script>
    new Maskfy('[data-mask]');
</script>

JS init by Class HTML

<script>
    new Maskfy('.mask-by-class');
</script>

JS Options

<script>
    new Maskfy({
        tag: '.mask-by-class', // selector tag
        reverse: true, // reverse typing
        mask: '999.999.999,99', // mask pattern
        size: 3 // minimum digits
    });
</script>
